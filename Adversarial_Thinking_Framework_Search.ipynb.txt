        "# Cell 6: serper.dev search client\n",
        "# Spec: returns List[SearchResult], raises typed errors, max 10 results\n",
        "\n",
        "async def serper_search(query: str, *, max_results: int = 10) -> List[SearchResult]:\n",
        "    \"\"\"\n",
        "    Calls serper.dev Search API.\n",
        "    Preconditions : query non-empty, SERPER_API_KEY set.\n",
        "    Postconditions: list length <= max_results, each entry validated.\n",
        "    Raises        : SerperAuthError, SerperRateLimit, SerperTimeout, SerperNetworkError\n",
        "    \"\"\"\n",
        "    if not query.strip():\n",
        "        raise ValueError(\"query must be non-empty\")\n",
        "\n",
        "    url = \"https://google.serper.dev/search\"\n",
        "    headers = {\"X-API-KEY\": SERPER_API_KEY, \"Content-Type\": \"application/json\"}\n",
        "    payload = {\"q\": query, \"num\": max_results}\n",
        "\n",
        "    try:\n",
        "        async with httpx.AsyncClient(timeout=15.0) as client:\n",
        "            resp = await client.post(url, json=payload, headers=headers)\n",
        "\n",
        "        if resp.status_code == 401:\n",
        "            raise SerperAuthError(f\"Serper auth failed: {resp.text}\")\n",
        "        if resp.status_code == 429:\n",
        "            raise SerperRateLimit(\"Serper rate limit hit\")\n",
        "        resp.raise_for_status()\n",
        "\n",
        "        data = resp.json()\n",
        "        results = []\n",
        "        for i, item in enumerate(data.get(\"organic\", [])[:max_results], start=1):\n",
        "            try:\n",
        "                results.append(SearchResult(\n",

        "    except httpx.TimeoutException:\n",
        "        raise SerperTimeout(\"Serper request timed out\")\n",
        "    except (SerperAuthError, SerperRateLimit):\n",

        "    except Exception as e:\n",

        "# Cell 8: Pure, order-independent reducers\n",
        "# Spec: pure, no side effects, raises ReducerError only\n",
        "\n",
        "def reducer_max_confidence(\n",
        "    prev: List[SearchResult],\n",
        "    incoming: List[SearchResult]\n",
        ") -> List[SearchResult]:\n",
        "    \"\"\"\n",
        "    Merge strategy: keep the result with higher rank (lower rank number = better).\n",
        "    Order-independent: result is the same regardless of which list comes first.\n",
        "    \"\"\"\n",
        "    try:\n",
        "        combined: Dict[str, SearchResult] = {}\n",
        "        for r in prev + incoming:\n",
        "            key = str(r.url)\n",
        "            if key not in combined or r.rank < combined[key].rank:\n",
        "                combined[key] = r\n",
        "        # Re-assign sequential ranks, sorted by original rank\n",
        "        sorted_results = sorted(combined.values(), key=lambda x: x.rank)\n",
        "        return [\n",
        "            SearchResult(title=r.title, url=r.url, snippet=r.snippet, rank=i+1)\n",
        "            for i, r in enumerate(sorted_results)\n",
        "        ]\n",
        "    except Exception as e:\n",
        "        raise ReducerError(f\"max_confidence merge failed: {e}\") from e\n",
        "\n",
        "\n",
        "def reducer_append_unique(\n",
        "    prev: List[WebPageContent],\n",
        "    incoming: List[WebPageContent]\n",
        ") -> List[WebPageContent]:\n",
        "    \"\"\"\n",
        "    Merge strategy: append items from incoming that aren't already in prev (by URL).\n",
        "    Order-independent: dedup is key-based, not position-based.\n",
        "    \"\"\"\n",
        "    try:\n",
        "        seen = {str(p.url) for p in prev}\n",
        "        merged = list(prev)\n",
        "        for item in incoming:\n",
        "            if str(item.url) not in seen:\n",
        "                merged.append(item)\n",
        "                seen.add(str(item.url))\n",
        "        return merged\n",
        "    except Exception as e:\n",
        "        raise ReducerError(f\"append_unique merge failed: {e}\") from e\n",
        "\n",
        "\n",
        "def reducer_merge(\n",
        "    prev: Dict[str, Any],\n",
        "    incoming: Dict[str, Any],\n",
        "    key: str,\n",
        "    strategy: Literal[\"max_confidence\", \"append_unique\", \"overwrite\"]\n",
        ") -> Dict[str, Any]:\n",
        "    \"\"\"\n",
        "    Unified reducer dispatcher — matches spec interface contract exactly.\n",
        "    \"\"\"\n",
        "    result = dict(prev)\n",
        "    try:\n",
        "        if strategy == \"max_confidence\":\n",
        "            result[key] = reducer_max_confidence(prev.get(key, []), incoming.get(key, []))\n",
        "        elif strategy == \"append_unique\":\n",
        "            result[key] = reducer_append_unique(prev.get(key, []), incoming.get(key, []))\n",
        "        elif strategy == \"overwrite\":\n",
        "            result[key] = incoming.get(key, prev.get(key))\n",
        "        else:\n",
        "            raise ReducerError(f\"Unknown strategy: {strategy}\")\n",
        "    except ReducerError:\n",
        "        raise\n",
        "    except Exception as e:\n",
        "        raise ReducerError(f\"reducer_merge failed on key={key}: {e}\") from e\n",
        "    return result\n",
        "\n",
        "print(\"✅ Reducers defined\")"

        "# Cell 11: SearchNode\n",
        "# Spec: reads user_query, writes search_results\n",
        "# Error handling: auth→abort, rate limit→backoff 3x, timeout→empty+continue\n",
        "\n",
        "def search_node(state: dict) -> dict:\n",
        "    \"\"\"Calls serper.dev with exponential backoff on rate limit.\"\"\"\n",
        "    query = state[\"user_query\"]\n",
        "    errors = list(state.get(\"errors\", []))\n",
        "\n",
        "    async def _run():\n",
        "        for attempt in range(4):   # 1 + 3 retries\n",
        "            try:\n",
        "                results = await serper_search(query)\n",
        "                log.info(f\"[SearchNode] Got {len(results)} results\")\n",
        "                return {\"search_results\": [r.model_dump() for r in results]}\n",
        "            except SerperAuthError as e:\n",
        "                log.error(f\"[SearchNode] Auth error: {e}\")\n",
        "                errors.append(str(e))\n",
        "                return {\"search_results\": [], \"errors\": errors}\n",
        "            except SerperRateLimit as e:\n",
        "                wait = 2 ** attempt\n",
        "                log.warning(f\"[SearchNode] Rate limit, waiting {wait}s (attempt {attempt+1})\")\n",
        "                await asyncio.sleep(wait)\n",
        "                if attempt == 3:\n",
